#version 330 core

// Thuộc tính đầu vào từ rlgl
in vec3 vertexPosition;
in vec2 vertexTexCoord;
in vec3 vertexNormal;
in vec4 vertexColor;

// Uniforms mặc định của Raylib
uniform mat4 mvp;
uniform mat4 matModel;

// Truyền xuống Fragment Shader
out vec3 fragPosition;
out vec2 fragTexCoord;
out vec3 fragNormal;
out vec4 fragColor;

void main()
{
    // Tính toán vị trí thực tế trong thế giới 3D (World Space)
    fragPosition = vec3(matModel * vec4(vertexPosition, 1.0));
    
    fragTexCoord = vertexTexCoord;
    
    // Xoay vector pháp tuyến theo mô hình 3D
    fragNormal = normalize(vec3(matModel * vec4(vertexNormal, 0.0)));
    
    // Nhận màu gốc (lệnh rlColor4ub(255,255,255,255) bên file C)
    fragColor = vertexColor;
    
    // Tính toán tọa độ hiển thị trên màn hình
    gl_Position = mvp * vec4(vertexPosition, 1.0);
}
